# Installation

    npm install uuid-generator-ts

# Usage

    import {UUID} from 'uuid-generator-ts';

    const uuid = new UUID();

    // Get a dash free UUID as string 'xxxxxxxxxxxx4xxxyxxxxxxxxxxxxxxx'
    uuid.getDashFreeUUID();
